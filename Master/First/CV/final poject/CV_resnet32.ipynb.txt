        "# Supported image extensions\n",
        "IMG_EXTENSIONS = ('.png', '.jpg', '.jpeg')\n",
        "\n",
        "# Painter code mapping (filename prefix → painter name)\n",
        "PAINTER_CODE_MAP = {\n",
        "    'Ba': 'Baleison',\n",
        "    'Bi': 'Biazaci',\n",
        "    'C':  'Canavesio',\n",
        "    'Ca': 'Canavesio',   # some files may use 'Ca'\n",
        "    'P':  'Pietro Guido',\n",
        "}\n",
        "\n",
        "def extract_letter_from_folder(folder_name: str) -> str:\n",
        "    \"\"\"\n",
        "    Extract the letter label from the folder name.\n",
        "    Example: 'Lettre_A' -> 'A'.\n",
        "    \"\"\"\n",
        "    return folder_name.split('_')[-1]\n",
        "\n",
        "def extract_painter_from_filename(filename: str) -> str:\n",
        "    \"\"\"\n",
        "    Extract the painter name from an image filename.\n",
        "    Rule:\n",
        "      - Remove leading '-' if present.\n",
        "      - Take the first token (up to an underscore or dash).\n",
        "      - If the token is a known painter code, return the painter name.\n",
        "      - Otherwise, return 'Inconnus' (unknown).\n",
        "    \"\"\"\n",
        "    name = filename\n",
        "    if name.startswith('-'):\n",
        "        name = name[1:]\n",
        "    token = name.split('_')[0].split('-')[0]\n",
        "    return PAINTER_CODE_MAP.get(token, 'Inconnus')   # fallback is Inconnus\n",
        "\n",
        "def load_dataset(data_dir: Path) -> pd.DataFrame:\n",
        "    \"\"\"\n",
        "    Build a DataFrame with columns:\n",
        "      [image_path, letter, painter, width, height]\n",
        "    \"\"\"\n",
        "    records = []\n",
        "    for folder in sorted(data_dir.iterdir()):\n",
        "        if not folder.is_dir():\n",
        "            continue\n",
        "        # Only consider folders starting with 'Lettre_'\n",
        "        if not folder.name.startswith('Lettre_'):\n",
        "            print(f\"  [Loader] Skipping unrecognised folder: {folder.name!r}\")\n",
        "            continue\n",
        "        letter = extract_letter_from_folder(folder.name)\n",
        "        for img_path in folder.rglob('*'):\n",
        "            if img_path.suffix.lower() in IMG_EXTENSIONS:\n",
        "                painter = extract_painter_from_filename(img_path.name)\n",
        "                try:\n",
        "                    with Image.open(img_path) as im:\n",
        "                        width, height = im.size\n",
        "                except Exception:\n",
        "                    width, height = None, None\n",
        "                records.append({\n",
        "                    'image_path': img_path,\n",
        "                    'letter': letter,\n",
        "                    'painter': painter,\n",
        "                    'width': width,\n",
        "                    'height': height,\n",
        "                })\n",
        "    return pd.DataFrame(records)\n",

        "# Constants for preprocessing\n",
        "FIXED_SIZE = (128, 128)   # width x height\n",
        "\n",
        "def load_image(path):\n",
        "    \"\"\"Load image from path, returns BGR (OpenCV default).\"\"\"\n",
        "    img = cv2.imread(str(path))\n",
        "    if img is None:\n",
        "        raise FileNotFoundError(f\"Image not found at {path}\")\n",
        "    return img\n",
        "\n",
        "def to_grayscale(img):\n",
        "    \"\"\"Convert BGR image to grayscale.\"\"\"\n",
        "    return cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)\n",
        "\n",
        "def histogram_equalization(gray_img):\n",
        "    \"\"\"Apply global histogram equalization to enhance contrast.\"\"\"\n",
        "    return cv2.equalizeHist(gray_img)\n",
        "\n",
        "def clahe_equalization(gray_img, clip_limit=2.0, tile_grid_size=(8,8)):\n",
        "    \"\"\"Apply CLAHE (Contrast Limited Adaptive Histogram Equalization).\"\"\"\n",
        "    clahe = cv2.createCLAHE(clipLimit=clip_limit, tileGridSize=tile_grid_size)\n",
        "    return clahe.apply(gray_img)\n",
        "\n",
        "def resize_to_fixed(img, size=FIXED_SIZE):\n",
        "    \"\"\"\n",
        "    Resize image to fixed size (width, height) without preserving aspect ratio.\n",
        "    This is the simplest approach for feeding into a CNN.\n",
        "    \"\"\"\n",
        "    return cv2.resize(img, size, interpolation=cv2.INTER_AREA)\n",
        "\n",
        "def normalize_image(img):\n",
        "    \"\"\"Normalize pixel values to [0, 1].\"\"\"\n",
        "    return img.astype(np.float32) / 255.0"
      ]

        "### 3.3 Multi‑View Dataset (Pixel values in [0, 1])\n",
        "\n",
        "Each image yields `V` strong augmentations for self‑supervision, or one light augmentation for evaluation. **Images are converted to grayscale** (single channel) so that the ResNet‑34 receives a 1‑channel input. No ImageNet normalisation is applied – the encoder sees pixel values in [0, 1]."
      ]
    },
    {
      "cell_type": "code",
      "execution_count": null,
      "id": "part3_dataset_code",
      "metadata": {},
      "outputs": [],
      "source": [
        "# Augmentations for grayscale images (channel = 1)\n",
        "multi_view_transform = v2.Compose([\n",
        "    v2.RandomResizedCrop(128, scale=(0.08, 1.0)),\n",
        "    v2.RandomHorizontalFlip(),\n",
        "    v2.RandomApply([v2.GaussianBlur(kernel_size=7, sigma=(0.1, 2.0))]),\n",
        "    v2.ToImage(),   # PIL 'L' -> tensor shape (1, 128, 128)\n",
        "    v2.ToDtype(torch.float32, scale=True),  \n",
        "])\n",
        "\n",
        "light_transform = v2.Compose([\n",
        "    v2.Resize(128),\n",
        "    v2.CenterCrop(128),\n",
        "    v2.ToImage(),\n",
        "    v2.ToDtype(torch.float32, scale=True),   # → [0,1]\n",
        "])\n",
        "\n",
        "class PainterMultiViewDataset(Dataset):\n",
        "    \"\"\"Returns (views, (painter_id, letter_id)).\"\"\"\n",
        "    def __init__(self, dataframe, V=1, transform=None):\n",
        "        self.df = dataframe.reset_index(drop=True)\n",
        "        self.V = V\n",
        "        self.transform = transform\n",
        "        self.test_transform = light_transform\n",
        "\n",
        "    def __len__(self):\n",
        "        return len(self.df)\n",
        "\n",
        "    def __getitem__(self, idx):\n",
        "        row = self.df.iloc[idx]\n",
        "        img_path = row['image_path']\n",
        "        painter_id = int(row['painter_label'])\n",
        "        letter_id  = int(row['letter_label'])\n",
        "        try:\n",
        "            # Convert directly to grayscale (mode 'L')\n",
        "            img = Image.open(img_path).convert('L')\n",
        "        except Exception:\n",
        "            img = Image.new('L', (128, 128))   # grayscale fallback\n",
        "\n",
        "        if self.V > 1 and self.transform:\n",
        "            views = [self.transform(img) for _ in range(self.V)]\n",
        "            return torch.stack(views), (painter_id, letter_id)\n",

        "class SIGReg(nn.Module):\n",
        "    \"\"\"SIGReg regulariser for leJEPA – encourages a smooth latent distribution.\"\"\"\n",
        "    def __init__(self, knots=17, proj_dim=proj_dim):\n",
        "        super().__init__()\n",
        "        t = torch.linspace(0, 3, knots, dtype=torch.float32)\n",
        "        dt = 3 / (knots - 1)\n",
        "        weights = torch.full((knots,), 2 * dt, dtype=torch.float32)\n",
        "        weights[[0, -1]] = dt\n",
        "        window = torch.exp(-t.square() / 2.0)\n",
        "        self.register_buffer(\"t\", t)\n",
        "        self.register_buffer(\"phi\", window)\n",
        "        self.register_buffer(\"weights\", weights * window)\n",
        "        self.A_buf = torch.empty(proj_dim, 256, device=device)   # pre‑allocate\n",
        "\n",
        "    def forward(self, proj):\n",
        "        self.A_buf.normal_()\n",
        "        self.A_buf.div_(self.A_buf.norm(p=2, dim=0))\n",
        "        A = self.A_buf\n",
        "        x_t = (proj @ A).unsqueeze(-1) * self.t\n",
        "        err = (x_t.cos().mean(-3) - self.phi).square() + x_t.sin().mean(-3).square()\n",
        "        return (err @ self.weights).mean() * proj.size(-2)\n",
        "\n",
        "\n",
        "def build_ResNet34_encoder(out_dim=128):\n",
        "    \"\"\"ResNet‑34 for grayscale 128×128 input, output dim `out_dim`.\"\"\"\n",
        "    model = resnet34(weights=None)\n",
        "    # Replace first convolution: 3 channels → 1 channel\n",
        "    model.conv1 = nn.Conv2d(1, 64, kernel_size=7, stride=2, padding=3, bias=False)\n",
        "    # The original fc expects 512 features (after avgpool and flatten).\n",
        "    # We replace it with LayerNorm + Linear to produce `out_dim`.\n",
        "    model.fc = nn.Sequential(\n",
        "        nn.LayerNorm(512),\n",
        "        nn.Linear(512, out_dim)\n",
        "    )\n",
        "    return model\n",
        "\n",
        "\n",
        "class MultiTaskProbe(nn.Module):\n",
        "    \"\"\"Linear probe on frozen features to predict painter and letter.\"\"\"\n",
        "    def __init__(self, input_dim=proj_dim, hidden_dim=256, num_painters=num_painters, num_letters=num_letters):\n",
        "        super().__init__()\n",
        "        self.ln = nn.LayerNorm(input_dim)\n",
        "        self.shared = nn.Sequential(\n",
        "            nn.Linear(input_dim, hidden_dim),\n",
        "            nn.ReLU(),\n",
        "        )\n",
        "        self.painter_head = nn.Linear(hidden_dim, num_painters)\n",
        "        self.letter_head  = nn.Linear(hidden_dim, num_letters)\n",
        "\n",
        "    def forward(self, x):\n",
        "        x = self.ln(x)\n",
        "        shared = self.shared(x)\n",
        "        return self.painter_head(shared), self.letter_head(shared)"
      ]

        "        painter_y = labels[0].to(device, non_blocking=True)  # ground truth painter\n",
        "        letter_y  = labels[1].to(device, non_blocking=True)  # ground truth letter\n",
        "\n",

        "        assert V_batch == V, f\"Expected V={V} but got {V_batch}\"\n",
        "\n",
        "        # Flatten batch and view dimensions: (B*V, C, H, W)\n",
        "        views_flat = views.view(B * V, C, H, W)\n",
        "\n",

        "            emb = net(views_flat)                     # (B*V, proj_dim)\n",
        "\n",
        "            # Reshape to (V, B, proj_dim) for multi-view operations.\n",
        "            proj = emb.reshape(B, V, -1).transpose(0, 1)  # (V, B, proj_dim)\n",
        "\n",

        "            # Invariance loss: each view’s embedding should be close to the mean.\n",

        "            sigreg_loss = sigreg(proj)\n",
        "            jepa_loss = lamb * sigreg_loss + (1 - lamb) * inv_loss\n",

        "def evaluate_painter_per_letter(letter):\n",
        "    \"\"\"Return accuracy and F1 for painter classification using only images of `letter`.\"\"\"\n",
        "    sub = test_df[test_df['letter'] == letter]\n",
        "    if len(sub) == 0:\n",
        "        return 0.0, 0.0, 0\n",
        "    ds = PainterMultiViewDataset(sub, V=1, transform=None)\n",
        "    loader = make_loader(ds, batch_size=64, shuffle=False)\n",
        "    all_preds, all_labels = [], []\n",
        "    net.eval()\n",
        "    probe.eval()\n",
        "    with torch.inference_mode():\n",
        "        for views, labels in loader:\n",
        "            views = views.to(device, non_blocking=True)\n",
        "            painter_y = labels[0].to(device, non_blocking=True)\n",
        "            with autocast(device_type='cuda', dtype=torch.bfloat16, enabled=USE_AMP):\n",
        "                emb = net(views)\n",
        "                painter_logits, _ = probe(emb)\n",
        "            pred = painter_logits.argmax(1)\n",
        "            all_preds.extend(pred.cpu().numpy())\n",
        "            all_labels.extend(painter_y.cpu().numpy())\n",
        "    acc = np.mean(np.array(all_preds) == np.array(all_labels))\n",
        "    f1 = f1_score(all_labels, all_preds, average='weighted')\n",
